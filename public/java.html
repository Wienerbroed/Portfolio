Create greeting method 


import org.springframework.beans.factory.annotation.Value;
import org.springframework.web.bind.annotation.*;

import java.util.HashMap;
import java.util.Map;

@RestController
@RequestMapping("/api")
@CrossOrigin(origins = "*")
public class HelloVisitorController {

    @Value("${name:Developer}")
    private String name;

    @GetMapping("/helloVisitor")
    public Map helloVisitor(@RequestParam(required = false, defaultValue = "Visitor") String nameParam) {
        Map response = new HashMap<>();
        response.put("message", "Hello " + nameParam + " and welcome to " + name + "'s portfolio");
        return response;
    }
}
