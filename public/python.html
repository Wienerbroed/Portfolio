Creating a simple anagram checker

Check if words are anagrams

Imports 

	
from flask import Flask, request, jsonify

Setup app 

	
app = Flask(__name__)

Create route 

	
@app.route('/api/matching-words', methods=['POST'])

Make function for anagram checker 

	
def matching_inputs():
    data = request.get_json()

    word = data.get('word')
    word1 = data.get('word1')

    if not word or not word1:
        return jsonify({'error': 'Please insert word in both fields'}), 400

    sorted_word = ''.join(sorted(word.lower()))
    sorted_word1 = ''.join(sorted(word1.lower()))

    if sorted_word == sorted_word1:
        message = f"{word} is an anagram of {word1}"
    else:
        message = f"{word} is NOT an anagram of {word1}"

    return jsonify({'message': message})
